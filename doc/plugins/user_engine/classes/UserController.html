edit_roles()

Edit the roles for a given User object. A user typically shouldn’t be allowed to edit their own roles, since they could assign themselves as Admins and then do anything. Therefore, keep this method locked down as much as possible.

[Source]

    # File vendor/plugins/user_engine/app/controllers/user_controller.rb, line 64
64:   def edit_roles
65:     if (@user = find_user(params[:id]))
66:       begin
67:         User.transaction(@user) do
68:         
69:           roles = params[:user][:roles].collect { |role_id| Role.find(role_id) }
70:           # add any new roles & remove any missing roles
71:           roles.each { |role| @user.roles << role if !@user.roles.include?(role) }
72:           @user.roles.each { |role| @user.roles.delete(role) if !roles.include?(role) }
73: 
74:           @user.save
75:           flash[:notice] = "Roles updated for user '#{@user.login}'."
76:         end
